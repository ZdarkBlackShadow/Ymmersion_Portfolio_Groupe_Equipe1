package main

import (
    "fmt"
    "math/rand"
    "time"
)

func main() {
    var Find bool = false
    var tentative int = 1
    rand.Seed(time.Now().Unix())
    Num := rand.Intn(100)
    fmt.Println("Bienvenue dans le jeu de devinettes !\nJe pense à un nombre entre 1 et 99. Pouvez-vous le devinez ?")
    for !Find {
        var Data int
        fmt.Scanln(&Data)
        if Data < Num {
            fmt.Println("C'est plus grand")
        } else if Data == Num {
            fmt.Printf("Bravo ! Vous avez trouvé le code en %d tentatives\n", tentative)
            Find = true
        } else {
            fmt.Println("C'est plus petit")
        }
        tentative += 1
    }
}
